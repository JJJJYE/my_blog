ROC

1) Package pROC

pacman::p_load("pROC")                          

ac     <- UB.ted$Personal.Loan                              #  

pp     <- predict(UB.rf, newdata=UB.ted, type="prob")[,2]   # "yes"    

rf.roc <- roc(ac, pp, plot=T, col="red")                    # roc( ,  ))

auc <- round(auc(rf.roc), 3)                                # AUC 
legend("bottomright",legend=auc, bty="n")

detach(package:pROC)

abline(0,1, col="black")
