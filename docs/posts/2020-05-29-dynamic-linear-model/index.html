library(dlm)

model1 <- function(p){
  
  mod               <- dlmModPoly(2) +  # Local Trend Linear 
                       dlmModSeas(12) #  12  
  V(mod)            <- exp(p[1])
  diag(W(mod))[1:3] <- exp(p[2:4])

  return(mod)  
}
