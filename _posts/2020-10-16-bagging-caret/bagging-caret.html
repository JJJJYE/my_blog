3-2-2. ROC

1) Package pROC

pacman::p_load("pROC")

test.rf.prob <- predict(caret.gd.rf, newdata = UB.ted, type = "prob")  #  Training Data    Test Data     
test.rf.prob <- test.rf.prob[,2]                                       # "yes"   


ac           <- UB.ted$Personal.Loan                                   #  
pp           <- as.numeric(test.rf.prob)                               # "yes"   

rf.roc       <- roc(ac, pp, plot = T, col = "red")                     # roc( ,  ))

auc          <- round(auc(rf.roc),3)
legend("bottomright",legend = auc, bty = "n")

detach(package:pROC)


2) Package Epi

pacman::p_load("devtools", "Epi")
# install_version("etm", version = "1.1", repos = "http://cran.us.r-project.org")

3) Package ROCR

pacman::p_load("ROCR")                                                  

rf.pred <- prediction(test.rf.prob, UB.ted$Personal.Loan)    # prediction( ,  )     


rf.perf <- performance(rf.pred, "tpr", "fpr")                # performance(, "", "1-")                        
plot(rf.perf, col = "red")                                   # ROC Curve
abline(0,1, col = "black")

perf.auc        <- performance(rf.pred, "auc")               # AUC
auc             <- attributes(perf.auc)$y.values
legend("bottomright", legend = auc, bty = "n")
