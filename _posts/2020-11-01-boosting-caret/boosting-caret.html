ac           <- UB.ted$Personal.Loan                                     #  

pp           <- as.numeric(test.ada.prob)                                # "yes"   


ada.roc     <- roc(ac, pp, plot = T, col = "red")                        # roc( ,  ))

auc <- round(auc(ada.roc), 3)                                            # AUC
