pacman::p_load("xgboost",                               # For xgb.train
               "Matrix")                                # For sparse.model.matrix


# XGBoost    !# sparse.model.matrix :    
trainm       <- sparse.model.matrix(Personal.Loan ~. , # Personal.Loan Target  
                                    data=UB.trd)  
testm        <- sparse.model.matrix(Personal.Loan ~. , # Personal.Loan Target  
                                    data=UB.ted)  

# For xgb.train
train_matrix <- xgb.DMatrix(data=as.matrix(trainm), label=UB.trd$Personal.Loan)
test_matrix  <- xgb.DMatrix(data=as.matrix(testm), label=UB.ted$Personal.Loan)

# Parameters
xgb_params <- list(objective = "binary:logistic", # Target

                   gamma = 0,                     #      /     NA= 5,                 # Tree  NA= 1,          #  leaf node   /      NA= 1,                 #  Data    Sample / 1   DataNA= 1)                    # Regularization


watchlist <- list(train=train_matrix)

set.seed(100)
UB.xgb <- xgb.train(params = xgb_params,          # List   NA=train_matrix,            # xgb.DMatrix  NA= 50,                 # nrounds :    
                    watchlist = watchlist)
